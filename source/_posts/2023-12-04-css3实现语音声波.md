``` html
<view class="animation start">
  <view
    class="item"
    v-for="(w, idx) in waveStart"
    :class="'waveStart-' + w"
    :key="idx"
  ></view>
  <view class="item" v-for="(w, idx) in waveList" :class="'wave-' + w" :key="idx"></view>
  <view class="item" v-for="(w, idx) in waveEnd" :class="'waveStart-' + w" :key="idx"></view>
</view>

```
